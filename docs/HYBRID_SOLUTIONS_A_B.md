### Flujo de Datos

// PatrÃ³n: Fire UI update immediately, queue event for storage

class LiveFirstExecutor {
  async executeStep(step, context) {
    const startTime = Date.now();

    // 1. Notificar UI inmediatamente (no espera)
    this.emitToUI({
      type: 'STEP_STARTED',
      step: step.name,
      message: step.friendlyMessage
    });

    // 2. Ejecutar el paso
    const result = await step.execute(context);

    // 3. Notificar UI del resultado (no espera)
    this.emitToUI({
      type: 'STEP_COMPLETED',
      step: step.name,
      preview: this.summarize(result)
    });

    // 4. Encolar evento para storage (no bloquea)
    this.eventQueue.enqueue({
      type: 'STEP_COMPLETED',
      step: step.name,
      duration: Date.now() - startTime,
      result: result, // Datos completos
      context: this.snapshotContext(context)
    });

    return result;
  }
}
