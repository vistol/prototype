// hooks/useEventStream.js

import { useState, useEffect, useCallback, useRef } from 'react';

export function useEventStream(executionId) {
  const [events, setEvents] = useState([]);
  const [isConnected, setIsConnected] = useState(false);
  const [error, setError] = useState(null);
  const eventSourceRef = useRef(null);
  const autoScrollRef = useRef(true);

  // Conectar al stream
  useEffect(() => {
    if (!executionId) return;

    const eventSource = new EventSource(
      `/api/executions/${executionId}/stream`
    );

    eventSource.onopen = () => {
      setIsConnected(true);
      setError(null);
    };

    eventSource.onmessage = (e) => {
      const event = JSON.parse(e.data);
      const formattedEvent = formatEvent(event);

      setEvents(prev => [...prev, formattedEvent]);
    };

    eventSource.onerror = (e) => {
      setIsConnected(false);
      if (eventSource.readyState === EventSource.CLOSED) {
        setError('ConexiÃ³n cerrada');
      }
    };

    eventSourceRef.current = eventSource;

    return () => {
      eventSource.close();
    };
  }, [executionId]);

  // Formatear evento para UI
  const formatEvent = useCallback((rawEvent) => {
    const config = EVENT_TYPES[rawEvent.type] || EVENT_TYPES.GENERIC;
    const formatted = config.format(rawEvent.data);

    return {
      id: rawEvent.id,
      sequence: rawEvent.sequence,
      timestamp: rawEvent.timestamp,
      type: rawEvent.type,
      level: typeof config.level === 'function'
        ? config.level(rawEvent.data)
        : config.level,
      icon: typeof config.icon === 'function'
        ? config.icon(rawEvent.data)
        : config.icon,
      color: typeof config.color === 'function'
        ? config.color(rawEvent.data)
        : config.color,
      ...formatted,
      raw: rawEvent
    };
  }, []);

  // Controles
  const setAutoScroll = useCallback((value) => {
    autoScrollRef.current = value;
  }, []);

  const clearEvents = useCallback(() => {
    setEvents([]);
  }, []);

  const exportEvents = useCallback((format = 'json') => {
    if (format === 'json') {
      return JSON.stringify(events, null, 2);
    }
    // CSV, etc.
  }, [events]);

  return {
    events,
    isConnected,
    error,
    autoScroll: autoScrollRef.current,
    setAutoScroll,
    clearEvents,
    exportEvents,
    stats: {
      total: events.length,
      byLevel: {
        info: events.filter(e => e.level === 'info').length,
        success: events.filter(e => e.level === 'success').length,
        warning: events.filter(e => e.level === 'warning').length,
        error: events.filter(e => e.level === 'error').length
      }
    }
  };
}

export function EventList({
  events,
  viewMode,
  autoScroll,
  onEventClick,
  filters
}) {
  const listRef = useRef();

  // Auto-scroll al final cuando llegan nuevos eventos
  useEffect(() => {
    if (autoScroll && listRef.current) {
      listRef.current.scrollToItem(events.length - 1);
    }
  }, [events.length, autoScroll]);

  // Filtrar eventos
  const filteredEvents = events.filter(event => {
    if (!filters.levels.includes(event.level)) return false;
    if (filters.steps.length && !filters.steps.includes(event.step)) return false;
    if (filters.search && !matchesSearch(event, filters.search)) return false;
    return true;
  });

  // Altura de fila segÃºn modo
  const getRowHeight = (index) => {
    const event = filteredEvents[index];
    if (viewMode === 'compact') return 28;
    if (viewMode === 'expanded') return event.expandable ? 200 : 80;
    // Normal
    return event.details ? 80 : 40;
  };

  const Row = ({ index, style }) => {
    const event = filteredEvents[index];
    return (
      <div style={style}>
        <EventRow
          event={event}
          viewMode={viewMode}
          onClick={() => onEventClick(event)}
        />
      </div>
    );
  };

  return (
    <AutoSizer>
      {({ height, width }) => (
        <List
          ref={listRef}
          height={height}
          width={width}
          itemCount={filteredEvents.length}
          itemSize={viewMode === 'compact' ? 28 : 60}
          overscanCount={10}
        >
          {Row}
        </List>
      )}
    </AutoSizer>
  );

  MIN(timestamp) as started_at,
  MAX(timestamp) as completed_at,
